name: Update DGR Atom Feed

on:
  schedule:
    # every hour
    - cron:  '0 * * * *'
  workflow_dispatch:  # allows manual runs

jobs:
  build-and-deploy:
    runs-on: ubuntu-latest

    steps:
    # 1. Checkout the gh-pages branch (create it first via GitHub UI)
    - name: Checkout gh-pages
      uses: actions/checkout@v3
      with:
        ref: gh-pages
        fetch-depth: 0

    # 2. Set up Python
    - name: Set up Python
      uses: actions/setup-python@v4
      with:
        python-version: '3.x'

    # 3. Install dependencies
    - name: Install dependencies
      run: |
        python -m pip install --upgrade pip
        pip install -r requirements.txt

    # 4. Run scraper (writes feeds/dgr.xml)
    - name: Run scraper
      run: python scraper.py

    # 5. Commit & push updated feed
    - name: Commit and Push
      run: |
        git config user.name "github-actions[bot]"
        git config user.email "github-actions[bot]@users.noreply.github.com"
        git add feeds/dgr.xml
        # Only commit if there are changes
        git diff --quiet && echo "No changes to commit" || git commit -m "ci: update DGR feed"
        git push origin gh-pages
